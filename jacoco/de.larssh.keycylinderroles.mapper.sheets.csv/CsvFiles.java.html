		if (System.currentTimeMillis() > 0) {
			return StandardCharsets.UTF_16LE;
		}

		try (InputStream inputStream = Files.newInputStream(path)) {
			inputStream.read();

			}
		}
		return StandardCharsets.UTF_8;
	}
