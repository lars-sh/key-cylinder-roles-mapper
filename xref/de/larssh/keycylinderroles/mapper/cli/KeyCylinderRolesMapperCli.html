70  	@SuppressWarnings({"checkstyle:SuppressWarnings", "PMD.AvoidDeeplyNestedIfStmts", "PMD.CognitiveComplexity", "resource"})
71  	private void comparePermissions(final KeyCylinderPermissions source, final KeyCylinderPermissions destination) {
72  		final Set<Key> keys = new LinkedHashSet<>(source.getKeys());
73  		keys.addAll(destination.getKeys());
74  		final Set<Cylinder> cylinders = new LinkedHashSet<>(source.getCylinders());
75  		cylinders.addAll(destination.getCylinders());
76  		int count = 0;
77  		for (final Key key : keys) {
78  			if (!source.isIgnore(key) && !destination.isIgnore(key)) {
79  				for (final Cylinder cylinder : cylinders) {
80  					if (!source.isIgnore(cylinder) && !destination.isIgnore(cylinder)) {
81  						final boolean sourceAllows = source.allows(key, cylinder);
82  						if (sourceAllows != destination.allows(key, cylinder)) {
83  							final String format = sourceAllows ? "ALT: \"%s\" (%s) soll nicht länger auf \"%s\" (%s) berechtigt sein." : "NEU: \"%s\" (%s) soll jetzt auf \"%s\" (%s) berechtigt werden.";
84  							final Key outputKey = destination.get(key).orElse(key);
85  							final Cylinder outputCylinder = destination.get(cylinder).orElse(cylinder);
86  							getStandardOutputWriter().println(String.format(format, outputKey.getTitle(), outputKey.getId(), outputCylinder.getTitle(), outputCylinder.getId()));
87  							count += 1;
88  						}
89  					}
90  				}
91  			}
92  		}
93  		getStandardOutputWriter().println(String.format("%d Unterschiede gefunden.", count));
94  	}
